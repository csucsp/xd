\course{MA102. Cálculo I}{Obligatorio}{MA102}

\begin{justification}
Constituye una de las asignaturas que proporciona la base para otras asignaturas de Matemática, así como la 
herramienta indispensable para abordar un sin número de aplicaciones de la derivada y de la integral, en la 
resolución de problemas en los diversos campos de las ciencias y de la tecnología.

% This course introduces first concepts of Linear Algebra as well as of Numerical Methods
% with an emphasis on problem-solving with the Scilab open-source computational
% package. Mathematical theory is restricted to fundamentals whereas effective pure and
% applied problem-solving is privileged. In each topic, only a few methods of relevance to
% everyday Engineering are taught. Knowledgeability about these methods prepares
% students for searching more advanced alternatives if and when necessary.

\end{justification}

\begin{unit}{Limites y Continuidad}{Limites y Continuidad}{Lion2014matematica, espinoza2012analisis, stewart2013calculo}{18}{C1}
% begin{unit}{}{Introduction}{Anton,Chapra}{18}{C1}
  \begin{topics}
      \item Límites: Algebraicos y trigonométricos.
      \item Limites infinitos y al infinito. Asíntotas. 
      \item Funciones continuas.
   \end{topics}

   \begin{learningoutcomes}
      \item Calcular límites algebraicos y trigonométricos. [\Usage].
      \item Aplicar adecuadamente las propiedades de Límites.[\Usage].
      \item Reconocer los puntos de discontinuidad de una función. [\Usage].
   \end{learningoutcomes}
\end{unit}

\begin{unit}{La derivada y sus aplicaciones}{La derivada y sus aplicaciones}{Lion2014matematica,edward1996calculo,espinoza2012analisis,stewart2013calculo}{24}{C1}
% begin{unit}{}{Linear Algebra}{Anton,Chapra}{14}{C1}
   \begin{topics}
    \item Definición e interpretación geométrica. 
    \item Reglas de Derivación. 
    \item Regla de la cadena. 
    \item Derivadas de funciones trigonométricas y trascendentales. 
    \item Derivadas implícitas. 
    \item Derivadas de orden superior. Derivación Logarítmica. 
    \item El criterio de la primera derivada. Criterio de la segunda derivada. 
    \item Aplicaciones de la derivada: Razón de cambio, Razones Relacionadas, Problemas de Máximos y Mínimos.
    \end{topics}

   \begin{learningoutcomes}
      \item Conocer el significado geométrico de la derivada.[\Usage].
      \item Modelar situaciones problemáticas aplicando derivadas.[\Usage].
      \item Graficar funciones aplicando los criterios de la derivada.[\Usage].
      \end{learningoutcomes}
\end{unit}

\begin{unit}{Integrales y técnicas de integración}{Integrales y técnicas de integración}{Lion2014matematica,edward1996calculo,espinoza2012analisis,stewart2013calculo}{24}{C24}
% begin{unit}{}{Numerical methods}{Anton,Chapra}{22}{C24}
   \begin{topics}
    \item Integral definida, la antiderivada, teorema fundamental del cálculo. 
    \item Tablas de integrales y sustituciones simples.
    \item Integral por partes. Integral trigonométrica.
    \item Funciones racionales y descomposición en fracciones parciales. 
    \item Sustitución trigonométrica. 
    \end{topics}

   \begin{learningoutcomes}
      \item Identificar, explorar y aplicar estrategias para el cálculo de integrales indefinidas.[\Usage].
      \item Analizar y aplicar los conocimientos previos para resolver integrales impropias.[\Usage].
      \end{learningoutcomes}
\end{unit}

\begin{unit}{Aplicaciones de la Integraln}{Aplicaciones de la Integral}{Lion2014matematica,edward1996calculo,espinoza2012analisis,stewart2013calculo}{18}{C24}
% begin{unit}{}{Numerical methods}{Anton,Chapra}{22}{C24}
   \begin{topics}
    \item Áreas de regiones planas.
    \item Volumen de un sólido por el método de secciones transversales.
    \item Volumen de un sólido por el método de capas cilíndricas.
    \item Longitud de una curva y área de una superficie de revolución.
    \end{topics}

   \begin{learningoutcomes}
      \item Analizar y decidir el método para el cálculo de áreas limitadas por curvas.[\Usage].
      \item Analizar y decidir al método adecuado para el cálculo del volumen de un sólido de revolución. [\Usage].
      \end{learningoutcomes}
\end{unit}
