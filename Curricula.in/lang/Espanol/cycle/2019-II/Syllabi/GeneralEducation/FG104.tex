\course{FG104. Introducción a la Filosofía}{Obligatorio}{FG104}

\begin{justification}
El estudio de la Filosofía en la universidad, se presenta como un espacio de reflexión constante sobre el ser y el quehacer del ser humano en el mundo. Así mismo, proporciona las herramientas académicas necesarias para la adquisición del pensamiento formal y la actitud crítica frente a las corrientes relativistas que nos alejan de la Verdad.
La formación filosófica aporta considerablemente al cultivo de los saberes, capacidades y potencialidades humanas, de tal manera que facilita al ser humano encontrar el camino hacia la Verdad plena.
\end{justification}

\begin{goals}
\item Estructurar en los alumnos los fundamentos filosóficos buscando desarrollar en ellos las capacidades superiores de pensamiento, a través del estudio crítico de los contenidos temáticos, que les permita despertar la avidez por el saber, el buscar la Verdad y el conocer la realidad de manera objetiva, de tal forma que puedan orientar su vida con criterios claros y razonables.
\end{goals}

\begin{unit}{Antropología}{Antropología}{MestasH,gaycurso,GarciaCuadrado, Zanotti,PlatonF,TomasA,CasaubonNociones,AyllonJ,LucasL,ReydeCastro}{6}{C21}
\begin{topics}
      \item Definición, etimología, objeto material y formal.
      \item Antropología física, cultural y filosófica.
      \item Visiones reduccionistas: Fenomenismo psicológico, dualismo, monismo.
      \item Visión integral del ser humano: Potencias orgánicas y espirituales.
      \item Fin último del hombre: La felicidad.
      \item Unión hilemórfica entre el cuerpo y alma.
      \item Fundamento metafísico de la persona humana. Boecio - Santo Tomás.
\end{topics}

\begin{learningoutcomes}
      \item Distinguir las nociones fundamentales de la antropología realista [\Usage].
      \item Identificar críticamente los reduccionismos antropológicos [\Usage].
      \item Identificar y valorar al hombre como un ser personal [\Usage].
\end{learningoutcomes}
\end{unit}

\begin{unit}{Gnoseología}{Gnoseología}{MestasH,CasaubonNociones,llanognoseologia,Sanguinetti,Maritain,Hessen,Zanotti,Platon,Pieper}{6}{C20}
\begin{topics}
      \item Definición, etimología, objeto material y formal. Gnoseología y epistemología.
      \item El Proceso del conocimiento entre el sujeto cognoscente y el objeto cognoscible. (sentidos externos e internos, intelecto agente y pasivo, reflexión).
      \item La verdad: La verdad ontológica y lógica, estados subjetivos ante la verdad. Certeza, duda, opinión, error.
      \item Corrientes gnoseológicas: Dogmatismo, escepticismo, racionalismo, empirismo, criticismo, realismo, idealismo.
\end{topics}
\begin{learningoutcomes}
      \item Examinar las dificultades que plantea el conocimiento humano y las diversas soluciones que se dan a las mismas [\Usage].
	\item Explicar los conceptos fundamentales del realismo gnoseológico [\Usage].
\end{learningoutcomes}
\end{unit}

\begin{unit}{Ética}{Ética}{MestasH,AristotelesE,KomarE,TomasASuma,GuardiniEtica,Piper2,RodriguezL2006,simon1981moral,SanchezM}{6}{C20}
\begin{topics}
      \item Características generales: etimología, objeto material y formal. Moral y ética.
      \item El bien. El fin último. La felicidad. Fin subjetivo, fin objetivo. Las virtudes.
      \item Criterios y fuentes de moralidad.
      \item Ley moral: Ley eterna, ley natural, ley 	particular.
      \item Los actos del hombre y los actos humanos.
      \item Voluntad y libertad.
      \item Elementos del acto humano: advertencia y 	consentimiento.
      \item Valoración moral del acto humano: objeto, fin, circunstancia.
      \item Relativismo ético.
\end{topics}
\begin{learningoutcomes}
    \item Distinguir las nociones esenciales de la ética filosófica desde sus fundamentos [\Usage].
    \item Explicar el valor de la vida virtuosa y asumirla como camino a la felicidad [\Usage].
    \item Examinar la naturaleza de la moralidad humana [\Usage].
\end{learningoutcomes}
\end{unit}

\begin{unit}{Metafísica}{Metafísica}{MestasH,Gomez, Alvira,TomasASuma,AristotelesMetafisica,EudaldoF}{6}{C20}
\begin{topics}
      \item La metafísica como estudio del ser: 	etimología, objeto material y formal.
      \item La estructura metafísica del ente: Sustancia-Accidente. Materia-Forma. Acto-Potencia. Esencia-Acto de ser.
      \item Los trascendentales.
      \item La causalidad: Causa material, formal, eficiente, final.
      \item Argumentos a priori y a posteriori para 	probar la existencia de Dios: San Anselmo y Santo Tomás. Las cinco vías de Santo Tomás.
\end{topics}
\begin{learningoutcomes}
      \item Examinar las características fundamentales de la metafísica y valorar su primacía en el pensamiento filosófico [\Usage].
      \item Comprender las nociones metafísicas fundamentales [\Usage].
      \item Explicar la posibilidad de la existencia de Dios desde la Teología natural [\Usage].
\end{learningoutcomes}
\end{unit}
