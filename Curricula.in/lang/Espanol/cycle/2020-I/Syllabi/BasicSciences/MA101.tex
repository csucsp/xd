\course{MA101. Matemática II}{Obligatorio}{MA101}

\begin{justification}
La importancia que ha alcanzado el Álgebra y la Geometría, constituye el acervo indispensable de conocimientos que permiten 
la utilización de herramientas matemáticas que utiliza la Ingeniería en todas sus ramas, en el diseño de nuevos tipos de tecnología 
necesaria para el desarrollo urbano, económico y social.  El curso de Matemática II permitirá al alumno alcanzar las habilidades 
para la utilización de dichas herramientas.

% The course develops in students the skills to deal with models of science and engineering skills. In the first part
% of the course a study of the functions of several variables, partial derivatives, multiple integrals and an
% introduction to vector fields is performed. Then the student will use the basic concepts of calculus to model
% and solve ordinary differential equations using techniques such as Laplace transforms and Fourier series.
\end{justification}

\begin{goals}
  \item Adquirir el hábito de razonar matemáticamente desde el punto de vista algebraico y geométrico.
  \item Desarrollar sus capacidades de análisis, abstracción y generalización.
  \item Efectuar correctamente las operaciones que se realizan con los números reales.
  \item Manejar los conceptos básicos de un sistema coordenado bidimensional, operaciones con funciones y obtener reglas de correspondencia que expresen modelos matemáticos simples.
\end{goals}

\begin{unit}{Sistema de ecuaciones Lineales y Matrices}{Sistema de ecuaciones Lineales y Matrices}{Lion2013matematica,kolman1999algebra,grossman2012algebra,figueroa2007vectores,barbolla1998sanz}{24}{C1,C20}
% begin{unit}{}{Multi-Variable Function Differential}{Stewart,DennisZ}{24}{C1,C20}
   \begin{topics}      
    \item Sistemas de ecuaciones lineales: Eliminación gaussiana. Eliminación de Gauss- Jordán. Interpretación Geométrica. Teorema de Rango.
    \item Sistemas homogéneos
    \item Consistencia e inconsistencia de sistemas de ecuaciones lineales.
    \item Problemas de aplicación con solución de sistemas de ecuaciones lineales.
    \item Definición de matriz. Tipos.
    \item Álgebra de matrices: Adición, multiplicación por un escalar, multiplicación de matrices.
    \item Aplicaciones con operaciones matriciales.
    \item Determinantes y propiedades.
    \item Inversa de una matriz: Por Adjunta y Gauss- Jordán. Propiedades.
      \end{topics}

   \begin{learningoutcomes}
    \item Expresar un sistema de ecuaciones lineales en su forma matricial.[\Usage]
    \item Resolver un sistema de ecuaciones lineales, utilizando métodos de eliminación.[\Usage]
    \item Relacionar la intersección de rectas con la resolución de sistemas de ecuaciones.[\Usage]
    \item Analizar la consistencia e inconsistencia de sistemas de ecuaciones lineales.[\Usage]
    \item Resolver problemas envolviendo sistemas de ecuaciones lineales.[\Usage]
    \item Identificar y manipular los diferentes tipos de matrices.[\Usage]
    \item Manejar propiedades del álgebra de matrices.[\Usage]
    \item Calcular determinantes, aplicar propiedades de determinantes.[\Usage]
    \item Aplicar las matrices en problemas cotidianos, analizando e interpretando la solución.[\Usage]
    \item Aplicación de la inversa de una matriz en sistemas de ecuaciones lineales.[\Usage]
    \item Identificar y calcular la inversa de una matriz.[\Usage]
    \end{learningoutcomes}
\end{unit}

\begin{unit}{Geometría Analítica en $R^2$}{Geometría Analítica en $R^2$}{Lion2013matematica,kolman1999algebra,grossman2012algebra,figueroa2007vectores,lehman2012geometria}{24}{C1,C20}
% begin{unit}{}{Multi-Variable Function Differential}{Stewart,DennisZ}{24}{C1,C20}
   \begin{topics}      
    \item Definición de vector.
    \item Adición y multiplicación por un escalar. Representación geométrica. 
    \item Vectores paralelos.
    \item Norma de un vector. Vector unitario.
    \item El producto escalar. Vectores ortogonales. Ángulo entre vectores.
    \item Proyección y componente ortogonal.
    \item Ecuación de recta: Ecuación vectorial, paramétrica, simétrica, normal y general de la recta.
    \item Posiciones relativas de dos rectas: Rectas paralelas,perpendiculares e intersección de rectas.
    \item Distancia de un punto a una recta.
    \item Angulo entre dos rectas.
    \item Cónicas: La circunferencia, la parábola, la elipse y la hipérbola.
    \item Problemas de aplicación de cónicas.
      \end{topics}

   \begin{learningoutcomes}
    \item Identificar el concepto de vector, realizar operaciones con vectores e interpretarlos gráficamente.[\Usage]
    \item Diferenciar vectores paralelos y perpendiculares. Calcular la longitud de un vector.[\Usage]
    \item Establecer y manejar los conceptos de producto escalar y proyección ortogonal.[\Usage]
    \item Aplicar vectores en la solución de problemas geométricos.[\Usage]
    \item Formular y analizar las ecuaciones vectorial, normal y paramétrica de una recta.[\Usage]
    \item Reconocer rectas paralelas y perpendiculares.[\Usage]
    \item Resolver problemas de intersección de rectas.[\Usage]
    \item Reconocer las distintas expresiones algebraicas que representan a la misma recta e interpretar la información que brinda.[\Usage]
    \item Identificar la ecuación de una circunferencia, elipse, parábola e hipérbola.[\Usage]
    \end{learningoutcomes}
\end{unit}

\begin{unit}{Geometría Analítica en $R^3$}{Geometría Analítica en $R^3$}{Lion2013matematica,lehman2012geometria}{18}{C1,C20}
% begin{unit}{}{Multi-Variable function Integral}{Stewart,DennisZ}{24}{C1,C20}
  \begin{topics}
    \item El espacio $R^3$: El producto escalar, Norma de un vector, Vector unitario, Vectores paralelos y ortogonales, Ángulo entre vectores, Proyección y componente ortogonal, El producto vectorial, Propiedades.
    \item La recta y el plano en $R^3$.
    \item Posiciones relativas entre rectas y planos.
    \item Intersección de rectas y planos.
    \item Distancia: De un punto a una recta, de un punto a un plano y distancia de planos paralelos.
    \item Superfícies: Cilindros, Esfera, Elipsoide, Cono, Paraboloide e Hiperboloide.
    \item Aplicaciones de las superficies cuadráticas.
   \end{topics}
  
  \begin{learningoutcomes}
    \item Identificar el concepto de producto vectorial, propiedades y aplicaciones.[\Usage]
    \item Describir y analizar las ecuaciones de rectas y planos en el espacio.[\Usage]
    \item Resolver problemas de intersección de rectas y planos.[\Usage]
    \item Calcular distancias, ángulos e intersecciones entre rectas y planos.[\Usage]
    \item Reconocer las distintas expresiones algebraicas que representan al mismo plano en $R^3$ e interpretar la información que brinda.[\Usage]
    \item Reconocer que la intersección de una superficie cuadrática con un plano es una cónica.[\Usage]
    \end{learningoutcomes}

\end{unit}
