\course{FG104. Introducción a la Filosofía}{Obligatorio}{FG104}

\begin{justification}
La filosofía ayuda al estudiante a descubrir que la razón del ser humano esta abierta a la búsqueda de la verdad en las cosas de este mundo y a la iluminación de la Verdad,  se trata de aprender a ejercitar las capacidades de exploraciónm y compresión del mundo.  Esta disciplina es la plataforma que le permitirá comprender la realidad de manera profunda y estructurada, conocer la génesis de la ciencia, distinguir lo objetivo de lo subjetivo y al mismo tiempo aprender a cuestionarse sobre las causas ultimas y su incidencia en las diversas áreas de la cultura.
\end{justification}

\begin{goals}
\item Orientar al alumno para dar a conocer las razones de los eventos que lo circundan.
\item Identificar las tendencias clasicas y actuales en filosofía.
\end{goals}

\end{outcomes}

\begin{unit}{Preliminares}{Melendo,Pieper,JuanPabloA}{15}{2}
\begin{topics}
	\item La Cultura Occidental Las raíces de nuestra cultura: el aporte griego, romano y judeocristiano.
	\item Crisis de la cultura occidental.
	\item Las preguntas fundamentales de la existencia humana. ?`Es posible alcanzar la verdad?, el hombre como explorador.
	\item El origen de la filosofía:  las preguntas fundamentales y la necesidad de sentido.
	\item Noción de filosofía. Objeto material y formal.  Definición: Maritain.  Elementos básicos de lo que es y no es filosofía.
	\item Importancia de la reflexión filosófica ante la realidad. ?`Para que sirve la filosofía?.
	\item Necesidad de la filosofía para comprender en profundidad nuestras experiencias existenciales.  Respuesta a la pregunta ?`Quién soy?. Necesidad de la filosofía para dar fundamento sólido-antropológico y epistemológico- a las diversas disciplinas.
	\item La crisis de la verdad: Reduccionismo metodológico, Agnosticismo funcional, el subjetivismo y el relativismo, Mentalidad tecnologista.
\end{topics}
\begin{unitgoals}
	\item Que el alumno se aproxime a la filosofía, su concepto y problemas fundamentales, vislumbre el método y ejercicio propio de esta disciplina.
	\item Que el alumno comprenda la dimensión existencia de la filosofía.
	\item Aprender a categorizar las preguntas interiores y a plantearlas de una manera racional.
\end{unitgoals}
\end{unit}

\begin{unit}{Realismo Metafísico y Existencial}{Melendo,Pieper,JuanPabloA}{15}{2}
\begin{topics}
	\item El concepto de la verdad: \textit{adaequatio rei et intellectus}.
	\item En sí-en mí objetividad y subjetividad.
	\item La metafísica. Conceptos básicos: ser, ente, esencia, accidente, sustancia.
	\item Los principios del ser: identidad, no contradicción, inteligibilidad, causalidad, finalidad.
	\item El punto de partida de la reflexión filosófica: el realismo metafísico.
	\item La reflexión acerca de la realidad.
\end{topics}

\begin{unitgoals}
	\item Descubrir la constitución de la realidad  y las categorizaciones de la misma.
	\item Comprender los conceptos básicos de la metafísica y su importancia conceptual.
	\item Comprender la problemática de la verdad y el significado, sentido y razón de ser de ésta.
\end{unitgoals}
\end{unit}

\begin{unit}{Panorama Histórico de la Filosofía}{Melendo,Pieper,JuanPabloA,Chavez2003HistoriaDoctrinas}{15}{3}
\begin{topics}
	\item Breve aproximación histórica de la filosofía. El por qué del análisis histórico.
	\item El aporte de los griegos a la filosofía, Platón, Aristóteles.
	\item La riqueza de la mal llamada Edad media y la leyenda del oscurantismo, Santo Tomás de Aquino.
	\item La fe y la razón.
\end{topics}

\begin{unitgoals}
	\item Comprender los orígenes del pensamiento filosófico y conocer algunos de los hitos históricos y pensadores más importantes.
	\item Realizar un recorrido por la historia de la filosofía y conocer los aportes que esta disciplina ha dado a la cultura occidental.
	\item Comprender la crisis de la filosofía y el influjo de ésta en la cultura actual.
\end{unitgoals}
\end{unit}
