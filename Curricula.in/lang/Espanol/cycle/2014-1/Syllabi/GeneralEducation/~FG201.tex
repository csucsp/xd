\begin{syllabus}

\course{FG201. Artes Plásticas}{Electivo}{FG201}

\begin{justification}
El curso proporciona los elementos para poder apreciar las artes plásticas de una manera analítica y práctica, así como reconocer los hechos artísticos como resultado de valores humanos, estableciendo una comunicación a través de las épocas y las culturas.
\end{justification}

\begin{goals}
\item Dar al alumno un conocimiento práctico como teórico de las diferentes técnicas pictóricas.
\end{goals}

\begin{outcomes}
\ExpandOutcome{FH}{1}
\ExpandOutcome{HU}{1}
\end{outcomes}

\begin{unit}{Definiciones, Elementos y Principios de Diseño}{Rudolf,Arte}{9}{1}
\begin{topics}
	\item Qué es el Arte
	\item Estética
	\item Elementos de diseño, el punto, la línea, forma y espacio.
	\item Principios de Diseño, la unidad, el equilibrio, simetría y proporción
\end{topics}
\begin{unitgoals}
	\item Conocer las definiciones de Arte, Estética, así como los elementos que intervienen en una composición artística u obra, para que les permita a los alumnos hacer un análisis de la misma.
\end{unitgoals}
\end{unit}

\begin{unit}{Teoría del Color}{Rudolf,Arte}{6}{1}
\begin{topics}
	\item Introducción
	\item Historia
	\item La visión, la luz y la materia.
	\item Formación del color por adición, sustracción y partición. Circulo cromático.
	\item Dimensiones del color, tente, brillo y saturación.
	\item Psicología del color.
\end{topics}
\begin{unitgoals}
	\item Desarrollo de la sensibilidad del Color.
	\item Conocer las nuevas teorías del Color para una mejor apreciación del mundo que nos rodea.
\end{unitgoals}
\end{unit}

\begin{unit}{Historia del Arte}{Rudolf,Arte}{15}{1}
\begin{topics}
	\item La apreciación. La crítica y la historia del arte, modelos de apreciación.
	\item Historia del arte.
	\item Arte antiguo y arte clásico. Prehstoria, arte egipcio, arte griego y romano.
	\item Edad media. Arte bizantino, románico y gótico.
	\item Arte Precolombino. Arte preinca, arte en México, arte Maya.
	\item Arte Asiático de China y Japón
	\item Arte del Renacimiento
	\item Arte Barroco y virreynal en el Perú.
	\begin{subtopics}
		\item El Barroco, generalidades.
		\item El imaginario del Barroco en el Perú.
		\item Cultura de la Interioridad.
		\item Origenes y aportes del Barroco peruano.
		\item Arquitectura Barroca en el Perú.
	\end{subtopics}
	\item Rococó y Neoclásico
\end{topics}
\begin{unitgoals}
	\item Formular modelos de apreciación cualitativa haciendo uso de los conocimientos adquiridos
	\item.Conocer el desarrollo de las diferentes épocas analizando las manifestaciones artísticas para comprender las influencias de estas a través del tiempo.
\end{unitgoals}
\end{unit}

\begin{unit}{El siglo XIX, XX y XXI}{Rudolf,Arte}{12}{1}
\begin{topics}
	\item El siglo XIX, época de profundos cambios
	\begin{subtopics}
		\item Europa y América.
		\item La industrialización.
		\item El impresionismo, la fotografía y el cine.
	\end{subtopics}

	\item El Perú y los inicios de la República.
	\begin{subtopics}
		\item El costumbrismo.
		\item El paisajismo.
		\item Europa como modelo.
	\end{subtopics}
	\item Los albores del siglo XX
	\begin{subtopics}
		\item Europa y las vanguardias.
		\item La ENBA de Perú.
		\item El indigenismo en el Perú.
		\item La consolidación de loCosmopolita, la Abstraccion, el Pop Art, el Expresionismo y la Figura Fantástica.
		\item La década de la transición y dolor.
	\end{subtopics}
	\item El fin del siglo XX y el advenimiento del siglo XXI.
	\begin{subtopics}
		\item El Perú y América del Sur en convivencia con la ola mundial.
		\item El uso de otros lenguajes y el arte político.
	\end{subtopics}
\end{topics}

\begin{unitgoals}
	\item Conocer las manifestaciones artísticas a través del tiempo en nuestro país, como hermoso legado de nuestros antepasados y como base de nuestra cultura, además de compararla con lo que paso y pasa mundialmente.
\end{unitgoals}
\end{unit}

\begin{unit}{Exposiciones}{Rudolf,Arte}{3}{1}
\begin{topics}
	\item Trabajo de investigación, sobre el análisis de la obra de un artista peruano.
	\item Exposición del trabajo en grupos de 4 ó 5 personas.
\end{topics}
\begin{unitgoals}
	\item Realizar un trabajo de investigación, sobre el análisis de la obra de un artista peruano, que les permita a los alumnos no solamente conocer el trabajo como artistas, sino lo que han planteado en su obra como una innovación y como reflejo de su persona.
\end{unitgoals}
\end{unit}
