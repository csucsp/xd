\course{FG107. Antropología Filosófica y Teológica}{Obligatorio}{FG107}

\begin{justification}
El estudio de la antropología filosófica (o psicología filosófica) es necesario a la filosofía y a cualquier
profesión relacionada con las ciencias sociales en general; sobre todo si se tiene la intención de constituir
una cosmovisión. Ella constituye la base para la reflexión en ética y política, así como en los diversos
estudios de las ciencias del hombre (economía, derecho, sociología, psicología, pedagogía, etc.).
\end{justification}

\begin{goals}
\item Ser capaz de comprender la naturaleza humana (es decir, las facultades y las finalidades de cada facultad,
su jerarquización y posible dominio); la condición de persona humana y su dignidad.
\end{goals}

\begin{unit}{Estatuto científico de la antropología filosófica}{Estatuto científico de la antropología filosófica}{GarciaCuadrado,artigas2008filosofia}{3}{C20}

	\item Epistemología general:
	\begin{subtopics}
		\item Noción y objeto de la ciencia.
		\item Filosofía y ciencias particulares.
		\item La filosofía y las ciencias.
		\item División y Jerarquización de las ciencias.	
	\end{subtopics}

	\begin{subtopics}
		\item Diversas antropologías: física, cultural, filosófica.
		\item Reduccionismos y extrapolaciones ilícitas.
		\item Dificultades en el objeto.
		\item El término naturaleza, sus acepciones y dificultades.
		\item Antropología teológica.
	\end{subtopics}
\end{topics}

	\item Comprender la noción de naturaleza desde la filosofía clásica y su centralidad [\Usage].

\begin{unit}{Naturaleza humana I - El hombre: ANIMAL racional}{Naturaleza humana I - El hombre: ANIMAL racional}{GarciaCuadrado,verneaux1971filosofia}{12}{C20}
\begin{topics}
	\item El hombre como viviente: 
		\begin{subtopics} 
			\item La jerarquía del mundo natural: la vida como un «modo de ser. 
			\item Principio vital: el alma.
			\item Características de la vida.
			\item Tipos de alma.
			\item Operaciones vegetativas (operaciones básicas vitales).
		\end{subtopics}
	\item El hombre como viviente animal: 
		\begin{subtopics} 
			\item Semejanzas y diferencias con los vivientes:La sensibilidad interna y externa.Las tendencias sensibles: deseos (apetito concupiscible) e impulsos (apetito irascible).Pasiones (afectividad humana).
		\end{subtopics}
\end{topics}
\begin{learningoutcomes}
	\item Analizar y comprender la naturaleza humana utilizando una metodología ascendente: Comprender el fenómeno de la vida, las semejanzas y diferencias que tenemos con los vivientes como animales. [\Usage]
\end{learningoutcomes}

\begin{unit}{Naturaleza del hombre II - El hombre: animal RACIONAL}{Naturaleza del hombre II - El hombre: animal RACIONAL}{GarciaCuadrado}{9}{C10, C20}

	\item Facultades superiores en el hombre:
	\item La Inteligencia:
		\begin{subtopics}
			\item El objeto de la inteligencia.
			\item Verdad y conocimiento.
			\item Naturaleza de la inteligencia.

			\item Operaciones de la razón teórica y práctica.

		\end{subtopics}

			\item La naturaleza de la voluntad y objeto de la voluntad.

			\item Descripción del acto voluntario: volición, decisión y acción.

\begin{learningoutcomes}

\begin{unit}{La persona humana: fundamento, dignidad, derecho y bien común}{La persona humana: fundamento, dignidad, derecho y bien común}{GarciaCuadrado,xiberta2000introduccion,}{21}{C20}
\begin{topics}
	\item Persona:.
		\begin{subtopics}
			\item Origen de la noción de persona.
			\item La fundamentación metafísica de la persona humana.
			\item Otras aproximaciones a la fundamentación de persona.
			\item Dignidad de la persona humana.
		\end{subtopics}
	\item La libertad humana:
		\begin{subtopics}
			\item Pruebas de la libertad.
			\item Las formas (o sentidos) de la libertad.
			\item Naturaleza y límites de la libertad.
		\end{subtopics}
	\item El animal político:
		\begin{subtopics}
			\item Naturaleza social del hombre y naturaleza de la sociedad.
			\item El amor y la amistad.
		\end{subtopics}
	\item Sexualidad humana: hombre y mujer:
		\begin{subtopics}
			\item Concepciones antropológicas de la sexualidad humana
			\item Niveles de la sexualidad humana: Ideología de género.
			\item Naturaleza de la sexualidad (fines de la sexualidad humana).
			\item Matrimonio y familia en la sociedad.
		\end{subtopics}
\end{topics}
\begin{learningoutcomes}
	\item Analizar y comprender qué significa ser persona humana, desde los orígenes de la noción.[\Usage].
	\item Comprender la fundamentación de la dignidad de la persona humana, así como la fundamentación de los derechos humanos y la noción de bien común.[\Usage].
\end{learningoutcomes}
\end{unit}
