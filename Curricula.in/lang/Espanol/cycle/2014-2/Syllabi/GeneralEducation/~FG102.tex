\begin{syllabus}

\course{FG102. Metodología del Estudio}{Obligatorio}{FG102}

\begin{justification}
Los alumnos en formación profesional necesitan comprender la exigencia del trabajo universitario y dominar las variadas formas de estudiar, para que puedan seleccionar los métodos mas convenientes a su personal estilo de aprender y a la naturaleza de cada asignatura. De ese modo podrá aplicarlos a su trabajo universitario, haciendo exitoso se esfuerzo.
\end{justification}

\begin{goals}
\item Identificar técnicas de estudio adecuadas.
\item Reconocer métodos de investigación científica.
\end{goals}

\end{outcomes}

\begin{unit}{Primera Unidad}{Bernedo}{15}{1}
\begin{topics}
	\item La formación profesional y el profesional.
	\item La postura ante el reto del trabajo universitario.
	\item Organización personal y de los recursos par el estudio exitoso.
	\item La dosificación del tiempo y del Esfuerzo.
	\item Las características del trabajo e la Universidad y las exigencias del mismo.
	\item Planes personales del mismo.
\end{topics}
\begin{unitgoals}
	\item Comprender y aceptar la formación profesional como un proceso que exige trabajo intelectual consciente y organizado con enfoque el ejercicio profesional responsable.
	\item Explicar las características del trabajo universitario y los niveles de exigencia que esta circunstancia reclama.
\end{unitgoals}
\end{unit}

\begin{unit}{Segunda Unidad}{Flores}{15}{1}
\begin{topics}
	\item La ciencia.
	\item El método en general.
	\item Los métodos láicos.
	\item Conocimiento y aprendizaje
	\item Expresión y transmisión del conocimiento. Elementos básicos de la transmisión oral del conocimiento.
	\item Los recursos para el aprendizaje.
\end{topics}
\begin{unitgoals}
	\item Definir la ciencia y explicar el camino para su conocimiento, aprovechamiento. Definir conocimiento, aprendizaje, sus diferencias y caminos para su logro.
	\item Explicar lo que es la comprensión y ejercitarse en al expresión o transmisión del conocimiento comprendido.
\end{unitgoals}
\end{unit}

\begin{unit}{Tercera  Unidad}{Perez}{15}{1}
\begin{topics}
	\item Auto evaluación de recursos y uso de métodos.
	\item El subrayado y el resumen.
	\item La toma de apuntes.
	\item Los esquemas y cuadros sintácticos.
	\item Los mapas conceptuales.
	\item Los mapas mentales.
	\item Las inteligencias múltiples.
	\item La conducta asertiva.
	\item La resiliencia y el reto académico.
	\item El estilo personal de trabajo.
\end{topics}
\begin{unitgoals}
	\item Evidenciar conciencia y dominio aceptable de las principales técnicas de estudio, explicar y formular estrategias para las necesidades académicas.
	\item Definir el estilo personal de trabajo, en concordancia con sus características personales para el trabajo intelectual.
\end{unitgoals}
\end{unit}

\begin{unit}{Cuarta Unidad}{Blythe,Buzan,Goleman,FloresA}{15}{1}
\begin{topics}
	\item El estilo personal de estudio.
	\item Elección de recursos.
	\item La lectura.
	\item Los exámenes.
	\item El trabajo grupal y los roles en los grupos de trabajo.
	\item El trabajo de redacción.
	\item El aparto crítico.
	\item El articulo publicable.
\end{topics}
\begin{unitgoals}
	\item Aplicar adecuadamente las técnicas mas apropiadas a las necesidades del estudio, a la naturaleza del curso y a las exigencias del compromiso académico.
	\item Aplicar técnicas y recursos adecuados para el estudio grupal, los exámenes y la preparación de trabajos.
\end{unitgoals}
\end{unit}
